function	script	CWWR_SYS	{
	for (set .@i, 0; .@i < 22; set .@i, .@i + 1) {
		if (getarg(0) == 2)
			disablenpc "WOEW_"+.@i;
		else if (getarg(0) == 1 && !$WOEPLAY[.@i])
			enablenpc "WOEW_"+.@i;
		else if (getarg(0) == 0)
			disablenpc "WOEW_"+.@i;
		else if (getarg(0) == 3 && .@i > 0)
			disablenpc "WOEW_"+.@i;
	}
	return;
}
