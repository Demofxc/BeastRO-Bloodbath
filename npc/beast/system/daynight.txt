// สคริปต์สำหรับปรับโหมดกลางวันและกลางคืนสำหรับทุกแผนที่

-	script	DayNightController	-1,{
	end;
	
OnClock0600:
	// ตั้งค่าเป็นโหมดกลางวันสำหรับทุกแผนที่
	day;
	announce "ตอนนี้เป็นเวลากลางวันแล้ว!", bc_all, 0xFFFF00;
	end;

OnClock1900:
	// ตั้งค่าเป็นโหมดกลางคืนสำหรับทุกแผนที่
	night;
	announce "ตอนนี้เป็นเวลากลางคืนแล้ว!", bc_all, 0x0000FF;
	end;

OnInit:
	// ตรวจสอบเวลาเริ่มต้นเมื่อเซิร์ฟเวอร์เริ่มทำงาน
	if (gettime(3) >= 6 && gettime(3) < 19) {
		day;
	} else {
		night;
	}
	end;
}
